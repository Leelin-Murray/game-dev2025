function twoSum(nums, target) {
    const map = {};

    for (let i = 0; i < nums.length; i++) {
        const needed = target - nums[i];

        if (map.hasOwnProperty(needed)) {
            return [map[needed], i];
        }

        map[nums[i]] = i;
    }

    return [];
}
