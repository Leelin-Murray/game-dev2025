Two Sum – JavaScript Solution

Given an array nums and a value target, return the indices of the two numbers that add up to target. Assume exactly one valid solution exists. 


// Two Sum function
function twoSum(nums, target) {
    const map = {};

    for (let i = 0; i < nums.length; i++) {
        const need = target - nums[i];

        if (need in map) {
            return [map[need], i];
        }

        map[nums[i]] = i;
    }
}
