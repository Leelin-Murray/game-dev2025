Question: Create a function that takes an array of integers nums and returns how many numbers in the array are **even**.


My Answer


function countEvens(nums) {
  let count = 0;

  for (let i = 0; i < nums.length; i++) {
    if (nums[i] % 2 === 0) {
      count++;
    }
  }

  return count;
}

ChatGPT Answer


function countEvens(nums) {
  return nums.filter(n => n % 2 === 0).length;
}
